#for Hedromechanics

import matplotlib.pyplot as plt

X=1
m1=101.0
m=101
T=10
n=2001
a=1
bc1=0
bc2=0
CFL=0.5

k1=(m1-1)/4
k2=3*k1
k1=int(k1)
k2=int(k2)
h=X/(m1-1)
#print h
t=T/(n-1)
r=t*t/h/h
U_up=[[0]*m for i in range(n)]
U_lax=[[0]*m for i in range(n)]

#%%%%% Boundary conditions %%%%%

i=0
while i<n:
#   print i
    U_up[i][0]=bc1
    U_up[i][m-1]=bc2
    U_lax[i][0]=bc1
    U_lax[i][m-1]=bc2
    i=i+1
    if i>= n:
    break

#%%%%% Initial condition %%%%%
j=1
while j<=k1-1 :
    U_up[0][j]=0
    U_lax[0][j]=0
    j=j+1

j=k1
while j<=k2 :
    U_up[0][j]=1
    U_lax[0][j]=1
    j=j+1

j=k2
while j<=m-1:
    U_up[0][j]=0
    U_lax[0][j]=1
    j=j+1




#%%%%% Upwind scheme %%%%%
jarr=range(1,n-1)
iarr=range(1,m-1)
for j in jarr:
    for i in iarr:
        U_up[j][i]=CFL*U_up[j-1][i-1]+CFL*U_up[j-1][i]
        U_up[j][0]=U_up[j][m-1]
        U_up[j][0]=CFL*U_up[j-1][0]+CFL*U_up[j-1][m-1]


#%%%%% Lax-Wendroff %%%%%

jrr=range(1,n-1)
iarr=range(1,m-2)
for j in jarr:
    for i in iarr:
        U_lax[j][i]=U_lax[j-1][i]-CFL/2*(U_lax[j-1][0]-U_lax[j-1][m-2])+CFL**2/2*(U_lax[j-1][i+1]-2*U_lax[j-1][i-1])
    U_lax[j][m-1]=U_lax[j-1][m-1]-CFL/2*(U_lax[j-1][0]-U_lax[j-1][m-1])+CFL**2/2*(U_lax[j-1][0]-2*U_lax[j-1][m-1]+U_lax[j-1][m-2])
    U_lax[j][0]=U_lax[j][m-1]
